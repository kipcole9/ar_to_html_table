    def output_table_totals(options)
      return unless table_has_totals?
      html.tfoot do
        html.tr do
          first_column = true
          table_columns.each do |column| 
            value = first_column ? first_column_total(options) : totals[column[:name].to_s]
            output_cell_value(:th, value, column, options)
            first_column = false
          end
        end
      end    
    end
